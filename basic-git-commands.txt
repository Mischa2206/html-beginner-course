git init                      -> Initialize local git repo

git config --global user.name 'username'
git config --global user.email 'user@email'

git add <file>                -> Adds file(s) to staging area

git status                    -> Check status of working tree

git log                       -> Check Changelog
git checkout <commitID>       -> Checks out an older version          

git commit                    -> Commit changes in staging area
git commit -m 'changes'       

git push                      -> Push to remote repo
git pull                      -> Pull latest from remote repo

git clone                     -> clone remote repo locally

git branch <branch name>      -> Create a new branch
git checkout <branch name>    -> Switch to a branch
git checkout -b list-items    -> Create and switch to branch
git merge <branch name>       -> Merge a branch with another

clear                         -> Clears console
